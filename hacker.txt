local colors = {

    Color3.fromRGB(0, 255, 0),
    Color3.fromRGB(0, 0, 255),
    Color3.fromRGB(255, 255, 0),

}

local colorIndex = 1

local function cycleColors()
    while true do
        MainFrame.BackgroundColor3 = colors[colorIndex]
        ToggleButton.BackgroundColor3 = colors[colorIndex]
        RunButton.BackgroundColor3 = colors[colorIndex]
        for _, button in ipairs(FunctionButtonsFrame:GetChildren()) do
            if button:IsA("TextButton") then
                button.BackgroundColor3 = colors[colorIndex]
            end
        end
        colorIndex = colorIndex % #colors + 1
        wait(0.1)
    end
end

cycleColors()
